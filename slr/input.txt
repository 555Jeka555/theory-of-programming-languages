PROGRAM BubbleSort;
VAR
    arr: ARRAY[10] OF INT;
    i, j, temp: INT;
BEGIN

  arr[0] := 5;
  arr[1] := 2;

  i := 0;
  WHILE i < 9 LOOP
  BEGIN
    j := 0;
        WHILE j < 9 - i LOOP
        BEGIN
            IF arr[j] > arr[j+1] THEN
            BEGIN
                temp := arr[j];
                arr[j] := arr[j+1];
                arr[j+1] := temp;
            END;
        END;
    i := i + 1;
  END;
END.
